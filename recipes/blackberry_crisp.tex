\begin{recipe}{Blackberry Crisp}{6-8 servings}{45 min.}
  \ingredient[2]{cups}{blackberries (fresh!)}
  Preheat oven to 375\0F. Place blackberries in a greased 1-quart baking dish.
  \ingredient[2]{Tbsp}{sugar}
  \ingredient[1]{tsp}{corn starch}
  \ingredient[1\fr12]{tsp}{water}
  \ingredient[\fr12]{tsp}{lemon juice}
  In a small bowl, mix sugar, cornstarch, water, and lemon juice until smooth.
  Pour over berries.
  \ingredient[\fr12]{cup}{quick-cook oats}
  \ingredient[\fr14]{cup}{flour}
  \ingredient[\fr14]{cup}{brown sugar}
  \ingredient[\fr12-\fr34]{tsp}{cinnamon}
  \ingredient[\fr14]{cup}{butter (cold)}
  Combine the oats, flour, brown sugar, and cinnamon; cut in butter until
  crumbly. Sprinkle liberally over berries.
  \ingredient[$\infty$]{cup}{vanilla ice cream}
  Bake, uncovered, for 20-25 minutes or until filling is bubbly. Serve warm
  with ice cream.
\end{recipe}
