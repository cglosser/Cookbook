\begin{recipe}{White Chicken Chilli}{}{}
\ing[2]{}{rotisserie chickens}
Skin and debone chicken, cutting into small pieces.
\ing[1 \fr12]{tsp}{white pepper}
\ing[1]{tsp}{garlic powder}
\ing[1]{small}{onion, diced}
\ing[2]{cans}{Great Northern beans}
\ing[2]{cans}{chicken broth}
\ing[2]{cans}{chopped green chilies}
\ing[1]{can}{cream of chicken soup}
\ing[1]{can}{white corn}
\ing[3]{tbsp}{jalape\~{n}o pepper juice}
\ing[2]{tbsp}{fresh cilantro, minced}
\ing[1]{tsp}{salt}
Combine ingredients into 6 quart slow cooker. Cook on low for four to six hours.
Garnish with sour cream, mozzarella cheese, cilantro, or tortilla strips.
\end{recipe}
