\begin{recipe}{Banana Oat Cookies}{}{makes 6 to 12 cookies}
\ing[1]{mashed}{banana}
\ing[\fr14]{tsp}{ground cardamom}
\ing[\fr14]{tsp}{baking powder}
\ing[\fr14]{tsp}{cinnamon}
\ing[1]{pinch}{nutmeg}
\ing[1]{pinch}{cloves}
\ing[\fr14]{tsp}{vanilla extract}
Preheat oven to 340\degrees for $<$ \fr14 cup oats, or 365\degrees for $>$
\fr12 cup oats. Mash banana and add vanilla extract, cardamom, cinnamon,
nutmeg, cloves, and baking powder.
\ing[\fr14]{tsp}{instant coffee (optional)}
\ing[4-5]{pieces}{chocolate (optional)}
Add chocolate and coffee if desired.
\ing[\fr12]{cup}{rolled oats}
\ing[\fr14]{cup}{ground oats}
Fold in oats until just mixed -- do not over mix! The dough will feel
wet.\newstep

Spoon onto a parchment lined baking sheet and bake for 20-30 minutes until
cookies brown.

\end{recipe}
